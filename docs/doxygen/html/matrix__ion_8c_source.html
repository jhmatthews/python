436  {

439  rate_matrix[nn][mm] = 0.0;

444  /* The next block of loops populate the matrix. For simplicity of reading the code each process has its own loop. Some rates
445  actually dont change during each iteration, but those that depend on n_e will. All are dealt with together at the moment,
446  but this could be streamlined if it turns out that there is a bottleneck. */
447 
448  /* Now we populate the elements relating to PI depopulating a state */
449 
450 
451  for (mm = 0; mm < nions; mm++)
452  {
453  if (ion[mm].istate != ion[mm].z + 1) // we have electrons
454  {
455  rate_matrix[mm][mm] -= pi_rates[mm];
456  }
457  }

461

463  {
464  for (nn = 0; nn < nions; nn++)
465  {
466 
467  if (mm == nn + 1 && ion[nn].istate != ion[nn].z + 1 && ion[mm].z == ion[nn].z)
468  {
469  rate_matrix[mm][nn] += pi_rates[nn];
470  }
471  }
472  }
473 
474  /* Now we populate the elements relating to direct ionization depopulating a state */
475 
476  for (mm = 0; mm < nions; mm++)
477  {
478  if (ion[mm].istate != ion[mm].z + 1 && ion[mm].dere_di_flag > 0) // we have electrons and a DI rate
479  {
480  rate_matrix[mm][mm] -= (xne * di_coeffs[mm]);
481  }
482  }
483 
484  /* Now we populate the elements relating to direct ionization populating a state - this does depend on the electron density */
485 
486  for (mm = 0; mm < nions; mm++)
487  {
488  for (nn = 0; nn < nions; nn++)
489  {
490  if (mm == nn + 1 && ion[nn].istate != ion[nn].z + 1 && ion[mm].z == ion[nn].z && ion[nn].dere_di_flag > 0)
491  {
492  rate_matrix[mm][nn] += (xne * di_coeffs[nn]);
493  }

503  {
504  rate_matrix[mm][mm] -= xne * (rr_rates[mm] + xne * qrecomb_coeffs[mm]);

574  elements (which relate to other elements (in the chemicalsense) with zeros. This is equivalent to the equation

604  }
605  }

728  }
