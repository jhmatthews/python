184 convergence (xplasma)
185  PlasmaPtr xplasma;
186 {
187  int trcheck, techeck, hccheck, whole_check, converging;
188  double epsilon;
189 
190  trcheck = techeck = hccheck = converging = 0;
191  xplasma->trcheck = xplasma->techeck = xplasma->hccheck = 0; //NSH 70g - zero the global variables
192  epsilon = 0.05;
193 
194  /* Check the fractional change in tempperatature and if is less than
195  * epsiolong increment trcheck and techeck
196  */
197 
198  if ((xplasma->converge_t_r =
199  fabs (xplasma->t_r_old - xplasma->t_r) / (xplasma->t_r_old +
200  xplasma->t_r)) > epsilon)
201  xplasma->trcheck = trcheck = 1;
202  if (xplasma->t_e < TMAX)
203  {
204  if ((xplasma->converge_t_e =
205  fabs (xplasma->t_e_old - xplasma->t_e) / (xplasma->t_e_old +
206  xplasma->t_e)) > epsilon)
207  xplasma->techeck = techeck = 1;
208  if ((xplasma->converge_hc =
209  fabs (xplasma->heat_tot - xplasma->cool_tot) / fabs (xplasma->heat_tot + xplasma->cool_tot)) > epsilon)
210  xplasma->hccheck = hccheck = 1;
211  }
212  else //If the cell has reached the maximum temperature
213  {
214  xplasma->techeck = techeck = xplasma->hccheck = hccheck = 2; //we mark it as overlimit
215  }
216 
217 //110919 nsh modified line below to include the adiabatic cooling in the check that heating equals cooling
218 //111004 nsh further modification to include DR and compton cooling, now moved out of lum_tot
219 
220  /* Check whether the heating and colling balance to within epsilon and if so set hccheck to 1 */
221  /* 130722 added a fabs to the bottom, since it is now conceivable that this could be negative if
222  cool_adiabatic is large and negative - and hence heating */
223 
224 /* NSH 130711 - also changed to have fabs on top and bottom, since heating can now be negative!) */
225 
226 /* NSH 130725 - moved the hc check to be within the if statement about overtemp - we cannot expect hc to converge if we are hitting the maximum temperature */
227  /* whole_check is the sum of the temperature checks and the heating check */
228 
229  xplasma->converge_whole = whole_check = trcheck + techeck + hccheck;
230 
231  /* Converging is a situation where the change in electron
232  * temperature is dropping with time and the cell is oscillating
233  * around a temperature. If that is the case, we drop the
234  * amount by which the temperature can change in this cycle
235  */
236 
237  if (xplasma->dt_e_old * xplasma->dt_e < 0
238  && fabs (xplasma->dt_e) > fabs (xplasma->dt_e_old))
239  converging = 1;
240  xplasma->converging = converging;
241 
242  if (converging == 1)
243  { // Not converging
244  xplasma->gain *= 0.7;
245  if (xplasma->gain < 0.1)
246  xplasma->gain = 0.1;
247  }
248  else
249  {
250  xplasma->gain *= 1.1;
251  if (xplasma->gain > 0.8)
252  xplasma->gain = 0.8;
253  }
254 
255  return (whole_check);
256 }
