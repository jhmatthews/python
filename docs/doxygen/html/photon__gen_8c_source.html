187 double
188 populate_bands (f1, f2, ioniz_or_final, iwind, band)
189  double f1, f2;
190  int ioniz_or_final;
191  int iwind;
192  struct xbands *band;
193 
194 {
195  double ftot, frac_used, z;
196  int n, nphot, most;
197 
198 /* Get all of the band limited luminosities */
199  ftot = 0.0;
200 
201  for (n = 0; n < band->nbands; n++) // Now get the band limited luminosities
202  {
203  if (band->f1[n] < band->f2[n])
204  {
205  xdefine_phot (band->f1[n], band->f2[n], ioniz_or_final, iwind);
206  ftot += band->flux[n] = geo.f_tot;
207  }
208  else
209  band->flux[n] = 0.0;
210 
211  if (band->flux[n] == 0.0)
212  band->min_fraction[n] = 0;
213  }
214 
215 /* So now we can distribute the photons among the different bands */
216 
217  frac_used = 0;
218 
219  for (n = 0; n < band->nbands; n++)
220  {
221  band->nat_fraction[n] = band->flux[n] / ftot;
222  frac_used += band->min_fraction[n];
223  }
224 
225  nphot = 0;
226  z = 0;
227  most = 0;
228 
229 
230  for (n = 0; n < band->nbands; n++)
231  {
232  band->used_fraction[n] = band->min_fraction[n] + (1 - frac_used) * band->nat_fraction[n];
233  nphot += band->nphot[n] = NPHOT * band->used_fraction[n];
234  if (band->used_fraction[n] > z)
235  {
236  z = band->used_fraction[n];
237  most = n;
238  }
239  }

242 add a few more photons to the band with most photons already. It should only be a few, at most
243 one photon for each band.*/
244 
245  if (nphot < NPHOT)
246  {
247  band->nphot[most] += (NPHOT - nphot);
248  }

250  return (ftot);

725  double f1, f2; /* The freqency mininimum and maximum if a uniform distribution is selected */
726  int spectype; /*The spectrum type to generate: 0 is bb, 1 (or in fact anything but 0)
727  is uniform in frequency space */

729 {

733  {
734  Error ("photo_gen_star: iend %d > NPHOT %d\n", iend, NPHOT);
735  exit (0);
736  }
737  if (f2 < f1)
738  {
739  Error ("photo_gen_star: Cannot generate photons if freqmax %g < freqmin %g\n", f2, f1);
740  }
741  Log_silent ("photo_gen_star creates nphot %5d photons from %5d to %5d \n", nphot, istart, iend);
742  freqmin = f1;
743  freqmax = f2;
744  r = (1. + EPSILON) * r; /* Generate photons just outside the photosphere */
745  for (i = istart; i < iend; i++)
746  {

750  p[i].grid = 0;

753  p[i].nnscat = 1;
754

756  {
757  p[i].freq = planck (t, freqmin, freqmax);
758  }
759  else if (spectype == SPECTYPE_UNIFORM)
760  { /* Kurucz spectrum */
761  /*Produce a uniform distribution of frequencies */
762  p[i].freq = random_number(freqmin,freqmax); //Generate a random frequency - this will exclude freqmin,freqmax.
763 
764  }
765  else
766  {
767  p[i].freq = one_continuum (spectype, t, geo.gstar, freqmin, freqmax);
768  }
769 
770  if (p[i].freq < freqmin || freqmax < p[i].freq)
771  {
772  Error_silent ("photo_gen_star: phot no. %d freq %g out of range %g %g\n", i, p[i].freq, freqmin, freqmax);
773  }
774 
775  randvec (p[i].x, r);
776 
777  if (geo.disk_type == DISK_VERTICALLY_EXTENDED)
778  {
779  while (fabs (p[i].x[2]) < zdisk (r))
780  {
781  randvec (p[i].x, r);
782  }
783 
784 
785  if (fabs (p[i].x[2]) < zdisk (r))
786  {
787  Error ("Photon_gen: stellar photon %d in disk %g %g %g %g %g\n", i, p[i].x[0], p[i].x[1], p[i].x[2], zdisk (r), r);
788  exit (0);
789  }
790  }
791 
792  randvcos (p[i].lmn, p[i].x);
793  }
794  return (0);
795 }

866  for (r = rmin; r < rmax; r += dr)

869  ltot += t * t * t * t * (2. * r + dr);

885  The extra factor of two arises because the disk radiates from both of its sides.

899  }

901  {
902  emit = emittance_bb (freqmin, freqmax, t);
903  }
904 
905  (*ftot) += emit * (2. * r + dr);
906  }
907 
908  (*ftot) *= q1;
909 
910 
911  /* If *ftot is 0 in this energy range then all the photons come elsewhere, e. g. the star or BL */
912

914  {
915  Log_silent ("disk_init: Warning! Disk does not radiate enough to matter in this wavelength range\n");
916  return (ltot);
917  }
918 
919  /* Now find the boundaries of the each annulus, which depends on the band limited flux.
920  Note that disk.v is calculated at the boundaries, because vdisk() interporlates on
921  the actual radius. */
922 
923  disk.r[0] = rmin;
924  disk.v[0] = sqrt (G * geo.mstar / rmin);
925  nrings = 1;
926  f = 0;
927  for (r = rmin; r < rmax; r += dr)
928  {
929  t = teff (tref, (r + 0.5 * dr) / rmin);
930  log_g = log10 (geff (gref, (r + 0.5 * dr) / rmin));
931 
932  if (spectype > -1)
933  { // continuum emittance
934  emit = emittance_continuum (spectype, freqmin, freqmax, t, log_g);
935  }
936  else
937  {
938  emit = emittance_bb (freqmin, freqmax, t);
939  }
940 
941  f += q1 * emit * (2. * r + dr);
942  /* EPSILON to assure that roundoffs don't affect result of if statement */
943  if (f / (*ftot) * (NRINGS - 1) >= nrings)
944  {
945  if (r <= disk.r[nrings - 1])
946  {
947  r = disk.r[nrings - 1] * (1. + 1.e-10);
948  }
949 
950  disk.r[nrings] = r;
951  disk.v[nrings] = sqrt (G * geo.mstar / r);
952  nrings++;
953  if (nrings >= NRINGS)
954  {
955  Error_silent ("disk_init: Got to ftot %e at r %e < rmax %e. OK if freqs are high\n", f, r, rmax);
956  break;
957  }
958  }
959  }
960  if (nrings < NRINGS - 1)
961  {
962  Error ("error: disk_init: Integration on setting r boundaries got %d nrings instead of %d\n", nrings, NRINGS - 1);
963  exit (0);
964  }
965 
966 
967  disk.r[NRINGS - 1] = rmax;
968  disk.v[NRINGS - 1] = sqrt (G * geo.mstar / rmax);
969 
970 
971  /* Now calculate the temperature and gravity of the annulae */
972 
973  for (nrings = 0; nrings < NRINGS - 1; nrings++)
974  {
975  r = 0.5 * (disk.r[nrings + 1] + disk.r[nrings]);
976  disk.t[nrings] = teff (tref, r / rmin);
977  disk.g[nrings] = geff (gref, r / rmin);
978  }
979 
980  /* Wrap up by zerrowing other parameters */
981  for (nrings = 0; nrings < NRINGS; nrings++)
982  {
983  disk.nphot[nrings] = 0;
984  disk.nhit[nrings] = 0;
985  disk.heat[nrings] = 0;
986  disk.ave_freq[nrings] = 0;
987  disk.w[nrings] = 0;
988  disk.t_hit[nrings] = 0;
989  }
990 
991  geo.lum_disk = ltot;
992  return (ltot);
993 }
