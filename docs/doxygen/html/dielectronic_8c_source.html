105 int
106 compute_dr_coeffs (temp)
107  double temp;
108 {
109  int n, n1, n2;
110  double Adi, Bdi, T0, T1;
111  for (n = 1; n < nions + 1; n++)
112  {
113  if (ion[n].drflag == 0) //There are no dielectronic coefficients relating to this ion
114  {
115  dr_coeffs[n] = 0.0; //So set the coefficients to zero
116  }
117  else
118  {
119  n1 = ion[n].nxdrecomb; //Obtain an index into the drecomb structure for this ion's data
120  dr_coeffs[n] = 0.0; //Zero the coefficient
121  if (drecomb[n1].type == DRTYPE_BADNELL) //It is a badnell type coeefficient
122  {
123  for (n2 = 0; n2 < drecomb[n1].nparam; n2++) //Loop over the parameters
124  {
125  dr_coeffs[n] += (drecomb[n1].c[n2] * exp (-1 * (drecomb[n1].e[n2] / temp)));
126  }
127  dr_coeffs[n] *= pow (temp, -1.5);
128  }
129  else if (drecomb[n1].type == DRTYPE_SHULL) //A schiull type parameter
130  {
131  Adi = drecomb[n1].shull[0];
132  Bdi = drecomb[n1].shull[1];
133  T0 = drecomb[n1].shull[2];
134  T1 = drecomb[n1].shull[3];
135  dr_coeffs[n] = Adi * pow (temp, -1.5) * exp ((-1.0 * T0) / temp);
136  dr_coeffs[n] *= (1 + Bdi * exp ((-1.0 * T1) / temp));
137  }
138  else
139  {
140  Error ("Compute_dr_coeffs: Unknown DR data rtype for ion %i\n", n);
141  dr_coeffs[n] = 0.0;
142  }
143  }
144  }
145  return (0);
146 }
