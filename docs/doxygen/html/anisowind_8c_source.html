612 randwind_thermal_trapping (p, nnscat)
613  PhotPtr p;

621  /* find the wind pointer for the photon */

623 
624  /* we want to normalise our rejection method by the escape
625  probability along the vector of maximum velocity gradient.
626  First find the sobolev optical depth along that vector */
627  tau_norm = sobolev (one, p->x, -1.0, lin_ptr[p->nres], one->dvds_max);
628 
629  /* then turn into a probability. Note that we take account of
630  this in trans_phot before calling extract */
631  p_norm = p_escape_from_tau (tau_norm);
632 
633  /* Throw error if p_norm is 0 */

638  z = 0.0;
639 
640  /* JM 1406 -- we increment nnscat here, and it is recorded in the photon
641  structure. This is done because we actuall have to multiply the photon weight
642  by 1/mean escape probability- which is nnscat. this is done in trans_phot.c
643  before extract is called.
644  */
645  *nnscat = *nnscat - 1;
646 
647  /* rejection method loop, which chooses direction and also calculated nnscat */
648  while (ztest > z)
649  {

657  ztest = random_number(0.0,1.0) * p_norm;
