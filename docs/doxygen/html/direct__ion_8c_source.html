97  int nrec, imax, imin, i;
98 
99  imin = imax = -1; //Set to avoid compiler warning
100 
101  nrec = ion[nion].nxderedi; // find the correct coefficient via an index in the ion structure
102 
103 
104  t = (BOLTZMANN * t_e) / (dere_di_rate[nrec].xi * EV2ERGS);
105  scaled_t = 1.0 - ((log (2.0)) / (log (2.0 + t))); //The fit is in 'reduced temperature' units
106 
107  /* 1/t is (hnu) / (k t_e). when this ratio is >100 we return 0, see #197 The integral will fail if we try to do it*/
108  if ((1.0 / t) > ALPHABIG_DIRECT_ION)
109  return 0.0;
110 
111  if (scaled_t < dere_di_rate[nrec].temps[0]) //we are below the range of DI data data
112  {
113 
114  Log_silent ("compute_di_coeffs: Requested temp %e is below limit of data for ion %i(Tmin= %e)\n",
115  scaled_t, nion, dere_di_rate[nrec].temps[0]);
116  imax = 1;
117  imin = 0; //We will try to extrapolate
118  }
119 
120  else if (scaled_t >= dere_di_rate[nrec].temps[dere_di_rate[nrec].nspline - 1]) //we are above the range of GS data
121  {
122  Log_silent ("compute_di_coeffs: Requested temp %e is above limit (%e) of data for ion %i\n",
123  scaled_t, dere_di_rate[nrec].temps[dere_di_rate[nrec].nspline - 1], nion);
124  imax = dere_di_rate[nrec].nspline - 1;
125  imin = dere_di_rate[nrec].nspline - 2; //We will try to extrapolate.
126  }
127 
128  else //We are within the range of tabulated data
129  {
130  for (i = 0; i < dere_di_rate[nrec].nspline - 1; i++)
131  {
132  if (dere_di_rate[nrec].temps[i] <= scaled_t && scaled_t < dere_di_rate[nrec].temps[i + 1]) //We have bracketed the correct temperature
133  {
134  imin = i;
135  imax = i + 1; //Set the indices to straddle the required temperature
136  }
137  }
138  }
139 
140  //Interpolate in log space
141 
142  drdt =
143  ((dere_di_rate[nrec].rates[imax]) - (dere_di_rate[nrec].rates[imin])) / ((dere_di_rate[nrec].temps[imax]) -
144  (dere_di_rate[nrec].temps[imin]));
145  dt = ((scaled_t) - (dere_di_rate[nrec].temps[imin]));
146  rate = dere_di_rate[nrec].rates[imin] + drdt * dt;
147 
148  coeff = pow (t, -0.5) * pow (dere_di_rate[nrec].xi, -1.5) * rate; //perform the interpolation
149 
150  if (exp (-1.0 / t) < (1.0 / VERY_BIG)) //Check to make sure the exponential integral will give a sensible result
151  {
152  exp_int = 0.0; //If its too small, it will just be zero
153  }
154  else
155  {
156  exp_int = gsl_sf_expint_E1 (1.0 / t); //Evaluate the first exponential integral using gsl library.
157  }
158 
159  coeff *= exp_int; //Finalise the coefficient.
160 
161  return (coeff);
162 }

318 {
319  int nion;
320  double u0;
321  double gaunt, coeff;
322  double root_etemp;
323 
324  gaunt = 0.1; //for now - from Mihalas for hydrogen
325  u0 = cont_ptr->freq[0] * H_OVER_K / electron_temperature;
326  nion = cont_ptr->nion;
327 
328  /* when hnu / kT ratio is >100 we return 0, see #197 */
329  if (u0 > ALPHABIG_DIRECT_ION)
330  return (0.0);
331 
332  /* if ion[n].dere_di_flag == 1 then we have direct ionization data for this ion
333  only do this if it is the ground state */
334  if (ion[nion].dere_di_flag == 1 && config[cont_ptr->nlev].ilv == 1)
335  {
336  root_etemp = sqrt (electron_temperature);
337  coeff = 2.07e-16 / (root_etemp * root_etemp * root_etemp);
338 
339  /* We now multiply by the ratio of the multiplicity of the ground states.
340  Dere's data comes from a vast range of sources, some PI cross sections
341  include excited states, others dont. We will make the approximation
342  of using juast the ground state multiplicities. If there is an error
343  introduced here it will be small */
344  coeff *= ion[nion].g / ion[nion + 1].g;
345 
346  coeff *= exp (u0);
347  coeff *= q_ioniz_dere (nion, electron_temperature);
348 
349 
350  /* do a sane check here, as there's an exponential which could blow up */
351  if (sane_check (coeff))
352  {
353  Error ("q_recomb is %8.4e for ion %i at temperature %8.4e, setting to zero\n", coeff, nion, electron_temperature);
354  coeff = 0.0;
355  }
356  }
357  else
358  coeff = 0.0;
359
