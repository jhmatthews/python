345 double
346 r_draw_from_path_histogram (Wind_Paths_Ptr PathPtr)
347 {
348  double r_rand, r_total, r_bin_min, r_bin_rand, r_path, r_bin_max;
349  int i_path = -1;
350 
351  r_total = 0.0;
352 // r_rand = PathPtr->d_flux * rand () / MAXRAND; DONE
353  r_rand = PathPtr->d_flux * random_number(0.0,1.0);
354  i_path = -1;
355 
356  //printf("DEBUG: r_rand %g out of total %g\n",r_rand, PathPtr->d_flux);
357  while (r_rand > r_total)
358  {
359  r_total += PathPtr->ad_path_flux[++i_path];
360  }
361 
362  //Assign photon path to a random position within the bin.
363  r_bin_min = reverb_path_bin[i_path - 1];
364  r_bin_max = reverb_path_bin[i_path];
365 // r_bin_rand = (rand () / MAXRAND) * (r_bin_max - r_bin_min); DONE
366  r_bin_rand = random_number(0.0,1.0) * (r_bin_max - r_bin_min);
367  r_path = r_bin_min + r_bin_rand;
368  return (r_path);
369 }

494 int
495 wind_paths_evaluate_single (Wind_Paths_Ptr paths)
496 {
497  int i;
498  paths->d_flux = 0.0;
499  paths->d_path = 0.0;
500  paths->i_num = 0;
501 
502  for (i = 0; i < geo.reverb_path_bins; i++)
503  { //For each path bin, add its contribution to total flux & avg path
504  paths->d_flux += paths->ad_path_flux[i];
505  paths->i_num += paths->ai_path_num[i];
506  paths->d_path += paths->ad_path_flux[i] * (reverb_path_bin[i] + reverb_path_bin[i + 1]) / 2.0;
507  }
508 
509  //If there was any data in this cell, calculate avg. path
510  if (paths->i_num > 0)
511  paths->d_path /= paths->d_flux;
512 
513  return (0);

728 wind_paths_output_vtk (WindPtr wind, int ndom)
729 {
730  FILE *fopen (), *fptr;
731  char c_file[LINELENGTH];
732  int i, j, k, n, i_obs, i_cells, i_points;
733  double r_azi, r_inc, r_x, r_y, r_z, r_err;
734  PhotPtr p_test = calloc (sizeof (p_dummy), 1);
735  DomainPtr dom;
736 
737  //Get output filename
738  sprintf (c_file, "%s.%d.wind_paths.vtk", files.root, ndom);
739 
740  if ((fptr = fopen (c_file, "w")) == NULL)
741  { //If this file can't be opened, error out
742  Error ("wind_paths_output_vtk: Unable to open %s for writing\n", c_file);
743  exit (0);
744  }
745  Log ("Outputting wind path information to file '%s'.\n", c_file);
746 
747  dom = &zdom[ndom];
748 
749  //Setup the number of vertexes and cells within this mesh
750  //Notably, we skip the outside cell as it's empty
751  //For the spherical case, we generate theta and thi bins using the same resolution
752  switch (dom->coord_type)
753  {
754  case SPHERICAL:
755  i_cells = 1 * (dom->ndim - 1) * geo.reverb_angle_bins * geo.reverb_angle_bins;
756  i_points = 1 * dom->ndim * (geo.reverb_angle_bins + 1) * (geo.reverb_angle_bins + 1);
757  break;
758  case CYLIND:
759  i_cells = 2 * (dom->ndim - 1) * (dom->mdim - 1) * geo.reverb_angle_bins;
760  i_points = 2 * dom->ndim * dom->mdim * (geo.reverb_angle_bins + 1);
761  break;
762  default:
763  free (p_test);
764  fclose (fptr);
765  Error ("wind_paths_output_vtk: Mesh format not yet supported (CYLVAR/RTHETA)");
766  return (0);
767  }
768 
769  //Write out header
770  fprintf (fptr, "# vtk DataFile Version 2.0\n");
771  fprintf (fptr, "Wind file data\nASCII\n");
772 
773  //Write out positions of corners of each wind cell as vertexes
774  fprintf (fptr, "DATASET UNSTRUCTURED_GRID\n");
775  fprintf (fptr, "POINTS %d float\n", i_points);
776  if (dom->coord_type == SPHERICAL)
777  {
778  for (i = 0; i < dom->ndim; i++)
779  {
780  n = dom->nstart + i;
781  for (j = 0; j <= geo.reverb_angle_bins; j++)
782  {
783  r_inc = j * (PI / (double) geo.reverb_angle_bins) - PI / 2.0;
784 
785  for (k = 0; k <= geo.reverb_angle_bins; k++)
786  {
787  r_azi = k * (PI / (double) geo.reverb_angle_bins);
788  r_x = wind[n].x[0] * sin (r_inc) * cos (r_azi);
789  r_y = wind[n].x[0] * sin (r_inc) * sin (r_azi);
790  r_z = wind[n].x[0] * cos (r_inc);
791  fprintf (fptr, "%10.5g %10.5g %10.5g\n", r_x, r_y, r_z);
792  }
793  }
794  }
795  }
796  else
797  {
798  for (i = 0; i < dom->ndim; i++)
799  {
800  for (j = 0; j < dom->mdim; j++)
801  {
802  wind_ij_to_n (ndom, i, j, &n);
803  for (k = 0; k <= geo.reverb_angle_bins; k++)
804  {
805  r_azi = k * (PI / (double) geo.reverb_angle_bins);
806  r_x = wind[n].x[0] * cos (r_azi);
807  r_y = wind[n].x[0] * sin (r_azi);
808  fprintf (fptr, "%10.5g %10.5g %10.5g\n", r_x, r_y, wind[n].x[2]);
809  fprintf (fptr, "%10.5g %10.5g %10.5g\n", r_x, r_y, -wind[n].x[2]);
810  }
811  }
812  }
813  }
814  fprintf (fptr, "\n");
815 
816  //Write out the vertexes comprising each cell
