231 //OLD double ds_to_wind(pp,ndom) calculates the photon pathlength to the edge of the wind.
232 //OLD
233 //OLD Arguments:
234 //OLD PhotPtr pp;.
235 //OLD
236 //OLD
237 //OLD Returns:
238 //OLD
239 //OLD The distance to the nearest boundary of the wind and the domain for which
240 //OLD the boudary applies.
241 //OLD
242 //OLD
243 //OLD Description:
244 //OLD
245 //OLD Python defines the boundaries of the wind in term of the intersection
246 //OLD of a biconical flow (defined by an inner and outer windcone) and
247 //OLD an inner and outer radius centered on the star. If the user is interested
248 //OLD in a biconical flow, he/she will most likely set the radii so that they
249 //OLD do not truncate the wind. Similarly, by choosing parameters for the windcones
250 //OLD properly, one can assure that one has a completely spherical flow. However,
251 //OLD the user may also perversely set the parameters to that both the conical
252 //OLD flow boundaries and the min and maximum radius come into play.
253 //OLD
254 //OLD Usually, one would
255 //OLD define the two types of The two types of boundaries are
256 //OLD usually defined so that one includes the other and so the cones apply when one is
257 //OLD dealing with a CV type wind, while the inner and outer radius applies for a spherical
258 //OLD model. However this routine does not require this to be the case, since it just
259 //OLD calculates where the edges are.
260 //OLD
261 //OLD In any event, if you are inside the wind already ds_to_wind calculates the distance to the edge of the wind.
262 //OLD If you are outside, It will also be to the nearest edge.
263 //OLD
264 //OLD The routine distinguishes between two basic cases. Either the photon is already in the wind
265 //OLD in which case one looks for the nearest boundary, or the photon is not currently
266 //OLD in the wind and you have to translate it until it reaches the the boundary (or
267 //OLD VERY_BIG)
268 //OLD Notes:
269 //OLD There is no guarantee that you will still be in the region defined by the grid.
270 //OLD
271 //OLD 1802 -ksl - At present this routine for imported models this routine only deals with
272 //OLD cylindrical models. Additionally for imported models we skip all of the
273 //OLD uwd of wind_cones. This is inefficient, and needs to be corrected for
274 //OLD rtheta and spherical models which can easily be handled using wind cones.
275 //OLD
276 //OLD History:
277 //OLD 1997 ksl Coded and debugged as part of Python effort.
278 //OLD 98dec ksl Updated to add an inner and outer radial boundary.
279 //OLD 99oct ksl The existing version of this code appears to contain

284 //OLD and logged the error. There ought to be a better fix
285 //OLD involving not letting the photon get too far from
286 //OLD the WD but I am worried about interactions between
287 //OLD this routine and other portions of the code. ???

842 int
843 walls (p, pold, normal)
844  PhotPtr p, pold;
845  double *normal;
846 {
847  double r, rho, rho_sq;
848  double xxx[3];
849  double s, z;
850  double theta, phi;
851 
852  /* Check to see if the photon has hit the star. If so
853  * put the photon at the star surface and use that position
854  * to determine the normal to the surface, the assumption
855  * being that the star is located at the center of the
856  * coordiante grid.
857  */
858 
859  if ((r = dot (p->x, p->x)) < geo.rstar_sq)
860  {
861  s = ds_to_sphere (geo.rstar, pold);
862  stuff_phot (pold, p);
863  move_phot (p, s);
864  stuff_v (p->x, normal);
865  return (p->istat = P_HIT_STAR);
866  }
867 
868  /* Check to see if it has hit the disk.
869  *
870  * For a vertically extended disk these means checking whether
871  * we are inside the maximum radius of the disk and then lookng
872  * comparing the z position of z to the height of the disk at
873  * that point*/
874 
875  if (geo.disk_type == DISK_VERTICALLY_EXTENDED)
876  {
877  rho = sqrt (p->x[0] * p->x[0] + p->x[1] * p->x[1]);
878  if ((rho) < geo.diskrad && fabs (p->x[2]) <= (z = zdisk (rho)))
879  {
880  /* 0 here means to return VERY_BIG if one has missed the disk, something
881  * that should not happen
882  */
883 
884  s = ds_to_disk (pold, 0);
885  if (s <= 0)
886  {
887  Error
888  ("walls: The previous position %e %e %e was inside the disk, correcting by %e \n",
889  pold->x[0], pold->x[1], pold->x[2], s);
890  s = ds_to_disk (pold, 0);
891  }
892  else if (s == VERY_BIG)
893  {
894  Error
895  ("walls: Should not miss disk at this position %e %e %e\n",
896  pold->x[0], pold->x[1], pold->x[2]);
897  s = ds_to_disk (pold, 0);
898  }
899  stuff_phot (pold, p);
900  move_phot (p, s - DFUDGE);
901 
902  /* Finally, we must calculate the normal to the disk at this point */
903 
904  theta = atan ((zdisk (r * (1. + EPSILON)) - z) / (EPSILON * r));
905  phi = atan2 (p->x[0], p->x[1]);
906 
907  normal[0] = (-cos (phi) * sin (theta));
908  normal[1] = (-sin (phi) * sin (theta));
909  normal[2] = cos (theta);
910 
911  if (p->x[2] < 0)
912  {
913  normal[2] *= -1;
914  }
915 
916 
917  return (p->istat = P_HIT_DISK);
918  }
919  }
920  else if (geo.disk_type == DISK_FLAT && p->x[2] * pold->x[2] < 0.0)
921  { /* Then the photon crossed the xy plane and probably hit the disk */
922  s = (-(pold->x[2])) / (pold->lmn[2]);
923  if (s < 0)
924  {
925  Error ("walls: distance %g<0. Position %g %g %g \n", s, p->x[0],
926  p->x[1], p->x[2]);
927  return (-1);
928  }
929  /* Check whether it hit the disk plane beyond the geo.diskrad**2 */
930  vmove (pold->x, pold->lmn, s, xxx);
