423 double
424 hydro_velocity (ndom,x, v)
425  int ndom;
426  double x[];
427  double v[];
428 {
429  double length ();
430  int ii, jj;
431  int im, jm;
432  double f1, f2;
433  double r, theta;
434  double v_r_interp, v_theta_interp, v_phi_interp;
435  double speed;
436  double xxx;
437 
438  if ((r = length (x)) == 0.0)
439  {
440  v[0] = v[1] = v[2] = 0.0;
441  return (0.);
442  };
443 
444 
445 
446  xxx = (sqrt (x[0] * x[0] + x[1] * x[1]) / r);
447  if (xxx >= 1.)
448  {
449  theta = PI / 2.;
450  }
451  else
452  theta = asin (xxx);
453 
454 
455  /* We compute the radial velocity - in zeus this is defined at the
456  cell centre in the theta dimension and the edge in the r dimension */
457 
458 
459  im = jm = ii = jj = 0;
460  f1 = f2 = 0.0;
461  hydro_frac (r, hydro_r_edge, ihydro_r, &im, &ii, &f1);
462  hydro_frac (theta, hydro_theta_cent, ihydro_theta, &jm, &jj, &f2);
463  v_r_interp = hydro_interp_value (v_r_input, im, ii, jm, jj, f1, f2);
464 
465 
466  /* We compute the theta velocity - in zeus this is defined at the
467  cell centre in the r dimension and the edge in the theta dimension */
468 
469  im = jm = ii = jj = 0;
470  f1 = f2 = 0.0;
471  hydro_frac (r, hydro_r_cent, ihydro_r, &im, &ii, &f1);
472  hydro_frac (theta, hydro_theta_edge, ihydro_theta, &jm, &jj, &f2);
473  v_theta_interp = hydro_interp_value (v_theta_input, im, ii, jm, jj, f1, f2);
474 
475 
476  /* We compute the phi velocity - in zeus this is defined at the
477  cell centre in both the r and theta dimension */
478 
479 
480  im = jm = ii = jj = 0;
481  f1 = f2 = 0.0;
482  hydro_frac (r, hydro_r_cent, ihydro_r, &im, &ii, &f1);
483  hydro_frac (theta, hydro_theta_cent, ihydro_theta, &jm, &jj, &f2);
484  v_phi_interp = hydro_interp_value (v_phi_input, im, ii, jm, jj, f1, f2);
485 
486 
487  /* Convert to carteisian velocity */
488 
489 
490  v[0] = v_r_interp * sin (theta) + v_theta_interp * cos (theta);
491  v[1] = v_phi_interp;
492  v[2] = v_r_interp * cos (theta) - v_theta_interp * sin (theta);
493 
494 
495  /*Compute the speed */
496 
497  speed = sqrt (v[0] * v[0] + v[1] * v[1] * v[2] * v[2]);
498 
499 
500 
501  if (sane_check (speed))
502  {
503  Error ("hydro_velocity:sane_check v %e %e %e\n", v[0], v[1], v[2]);
504  }
505  return (speed);
506 
507 }

702 rtheta_make_hydro_grid (w, ndom)
703  WindPtr w;
704  int ndom;
705 {
706  double theta, thetacen, dtheta;
707  int i, j, n;
708  int ndim, mdim;
709 
710  ndim = zdom[ndom].ndim;
711  mdim = zdom[ndom].mdim;
712 
713 
714  for (i = 0; i < ndim; i++)
715  {
716  for (j = 0; j < mdim; j++)
717  {
718  wind_ij_to_n (ndom, i, j, &n);
719  w[n].inwind = W_ALL_INWIND;
720  if (i == 0) // The inner edge of the grid should be geo.rstar
721  {
722  w[n].r = geo.rstar; //So we set the inner edge to be the stellar (or QSO) radius
723  w[n].rcen = (geo.rstar + hydro_r_edge[0]) / 2.0; //It will be a big cell
724  w[n].inwind = W_NOT_INWIND;
725  }
726  else if (i - 1 > ihydro_r) // We are at the radial limit of the data, this last cell will be a ghost cell of our own
727  {
728  w[n].r = geo.rmax; // so we set the outer cell to be the edge of the wind
729  w[n].rcen = geo.rmax; // And it has zero volume
730  w[n].inwind = W_NOT_INWIND;
731  }
732  else
733  {
734  w[n].r = hydro_r_edge[i - 1]; //The -1 is to take account of the fact that i=0 cell is the inner part of the geometry inside the wind
735  w[n].rcen = hydro_r_cent[i - 1];
736  }
737  dtheta = 2.0 * (hydro_theta_cent[j] - hydro_theta_edge[j]);
738 
739 
740 
741  if (hydro_theta_cent[j] + (dtheta / 2.0) > hydro_thetamax && hydro_theta_cent[j] - (dtheta / 2.0) < hydro_thetamax) //This cell bridges the boundary - we reset it so the lower edge is at the disk boundary
742  {
743  theta = hydro_theta_edge[j];
744  thetacen = ((hydro_thetamax + theta) / 2.0);
745  }
746 
747  else if (j > ihydro_theta) //We are setting up a cell past where there is any data
748  {
749  thetacen = hydro_thetamax; //Set the center and the edge to the maximum extent of the data/interest
750  theta = hydro_thetamax;
751  w[n].inwind = W_NOT_INWIND;
752  }
753  else
754  {
755  thetacen = hydro_theta_cent[j];
756  theta = hydro_theta_edge[j];
757  }
758  w[n].theta = theta * RADIAN;
759  w[n].thetacen = thetacen * RADIAN;
760  w[n].x[1] = w[n].xcen[1] = 0.0;
761  w[n].x[0] = w[n].r * sin (theta);
762  w[n].x[2] = w[n].r * cos (theta);
763  w[n].xcen[0] = w[n].rcen * sin (thetacen);
764  w[n].xcen[2] = w[n].rcen * cos (thetacen);
765 
766  }
767  }
768 
769  /* Now set up the wind cones that are needed for calclating ds in a cell */
770 
771  rtheta_make_cones (ndom, w);
772 
773 
774  /* OK finished successfuly */
