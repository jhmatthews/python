443  /* All the estimators need the volume so get that first. */
444  /* JM 1507 - we use cell volume for all the estimators here */
445 
446  volume = one->vol;
447 
448  /* bf estimators. During the mc calculation the quantity stored
449  was weight * cross-section * path-length / frequency.
450  To normalise this we need to put in:
451  1 / h (Planck constant)
452  1 / Volume
453  1 / Time
454  I think that the weights are chosen such that Time = 1.
455  So the normalisation of gamma and gamma_e is easy.
456  */
457 
458 
459  /* For the alpha_st estimators (stimulated recombination) the
460  estimator also needs to be multiplied by a temperature
461  dependent factor which is almost, but not quite, given by
462  the LTE population ratio. The muliplicative factor
463  is given by: */
464  stimfac = 0.5 * pow (H * H / 2. / PI / MELEC / BOLTZMANN / xplasma->t_e, 3. / 2.);
465 
466  for (i = 0; i < nlte_levels; i++)
467  {
468  for (j = 0; j < config[i].n_bfu_jump; j++)
469  {
470 
471  mplasma->gamma_old[config[i].bfu_indx_first + j] = mplasma->gamma[config[i].bfu_indx_first + j] / H / volume; //normalise
472  mplasma->gamma[config[i].bfu_indx_first + j] = 0.0; //re-initialise for next iteration
473  mplasma->gamma_e_old[config[i].bfu_indx_first + j] = mplasma->gamma_e[config[i].bfu_indx_first + j] / H / volume; //normalise
474  mplasma->gamma_e[config[i].bfu_indx_first + j] = 0.0; //re-initialise for next iteration
475 
476  /* For the stimulated recombination parts we need the the
477  ratio of statistical weights too.
478  For free electron statistical weight = 2 is included in
479  stimfac above. */
480 
481  stat_weight_ratio = config[phot_top[config[i].bfu_jump[j]].uplev].g / config[i].g;
482 
483  mplasma->alpha_st_old[config[i].bfu_indx_first + j] =
484  mplasma->alpha_st[config[i].bfu_indx_first + j] * stimfac * stat_weight_ratio / H / volume;
485  mplasma->alpha_st[config[i].bfu_indx_first + j] = 0.0;
486 
487  mplasma->alpha_st_e_old[config[i].bfu_indx_first + j] =
488  mplasma->alpha_st_e[config[i].bfu_indx_first + j] * stimfac * stat_weight_ratio / H / volume;
489  mplasma->alpha_st_e[config[i].bfu_indx_first + j] = 0.0;
490 
491  /* For continuua whose edges lie beyond freqmin assume that gamma
492  is given by a black body. */
493 
494  /* For now place the limit at 7.5e12 which is 400000AA */
495  /* Try also doing it for very high energy ones - greater than 50eV: 1.2e16 since up there the statistics of the estimators are very poor at the moment.
496  Ideally we don't want to have this so should probably switch this back sometime (SS August 05) !!!BUG */
497 
498  if (phot_top[config[i].bfu_jump[j]].freq[0] < 7.5e12 || phot_top[config[i].bfu_jump[j]].freq[0] > 5e18)
499  {
500  mplasma->gamma_old[config[i].bfu_indx_first + j] = get_gamma (&phot_top[config[i].bfu_jump[j]], xplasma);

504  }
505 
506  }
507

509 
510  /* For bb jumps the normalisation requires:
511  1/ 4 PI
512  1/ Volume
513  1/ Time
514  c / nu - to convert dvds^-1 to dnuds^-1
515 
516  Also, I'm putting the correction factor for stimulated emission in here - so that it doesn't have to be
517  computed in the macro atom jumping probabilities. (SS)
518 
519 
520  */
521 
522  for (j = 0; j < config[i].n_bbu_jump; j++)
523  {
524 
525  /* The correction for stimulated emission is (1 - n_lower * g_upper / n_upper / g_lower) */
526 
527  stimfac = den_config (xplasma, line[config[i].bbu_jump[j]].nconfigu) / den_config (xplasma, i);
528  stimfac = stimfac * config[i].g / config[line[config[i].bbu_jump[j]].nconfigu].g;

531  stimfac = 1. - stimfac; //all's well
532  }

638  density * ne * [(alpha_sp_e - alpha_sp) + (alpha_st_e - alpha_st)]

650  density = den_config (xplasma, cont_ptr->nlev);

654  total += cool_contribution;
655  }
656  }

659 
660  }
661 
662  return (total);
