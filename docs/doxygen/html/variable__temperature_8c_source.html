448  int nion;

457 
458  numerator = denominator = 0.0;
459  /*nion is the upper (recombining) ion in the pair, but calc_pi_rate uses the lower (ionizing) ion */
460  ion_lower = nion - 1;
461 
462 
463 
464  if (niterate == 0) /*first time of iterating this cycle, so calculate the numerator. This rate only depends on the model of J, which doesn't change during the iteration cycle */
465  {
466  if (mode == IONMODE_PAIRWISE_ML93)
467  {
468  numerator = calc_pi_rate (ion_lower, xplasma, 2, 1); /*Call calc_pi_rate with mode 2 (dilute BB), which returns the PI rate coeficient */
469  }
470  else if (mode == IONMODE_PAIRWISE_SPECTRALMODEL) /*Call calc_pi using a seriews a series of power laws and/or exponentials to represent the spectrum */
471  {
472  numerator = calc_pi_rate (ion_lower, xplasma, 1, 1);
473  }
474  xplasma->PWnumer[ion_lower] = numerator; /* Store the calculated numerator for this cell - it wont change during one ionization cycle */
475  } //End of if statement to decide if this is the first iteration
476  else
477  {
478  numerator = xplasma->PWnumer[ion_lower]; // We are on the second iteration, so use the stored value
479  }
480 
481  /*If the numeratior is 0, there is no PI for this ion, and so there
482  * is no need to waste time evaluationg the denominatory */
483  if (numerator == 0.0)
484  {
485  return (0.0);
486  }
487 
488 
489  if (pow (((xtemp - xplasma->PWdtemp[ion_lower]) / xplasma->PWdtemp[ion_lower]), 2) > 1e-6) //If our guess temperature hasnt changed much, use denominator from last time
490  {
491  /*calc_pi_rate uses the value of t_r and w in the plasma structure to compute the rate coefficient,
492  * so we need to store the actual value of t_r and w so we can substititute our guess temperature
493  * and w=1 (to get LTE) */
494 
495  t_r_store = xplasma->t_r;
496  w_store = xplasma->w;
497  xplasma->t_r = xtemp;
498  xplasma->w = 1.0;
499 
500  /*Now we can call calc_pi_rate, which will now calulate an LTE rate coefficient at our guess temperature */
501  denominator = calc_pi_rate (ion_lower, xplasma, 2, 1);
502 
503  /*Put things back the way they were */
504  xplasma->t_r = t_r_store;
505  xplasma->w = w_store;
506  xplasma->PWdenom[ion_lower] = denominator; /*Store the LTE rate coefficient for next time */
507  xplasma->PWdtemp[ion_lower] = xtemp; /*And also the temperature at which it was calculated */
508  }
509  else
510  {
511  denominator = xplasma->PWdenom[ion_lower]; /*The temperature hadnt changed much, so use the stored value */
512  }
513 
514 
515  q = numerator / denominator; /*This is the PI rate correction factor */
516 
517 
518  return (q);
519
