199  * in the rdpar statement, and then once we have the answer they are converted back to values that the
200  * program uses internally. Presumably this was done for historical reasons, having to do with running
201  * old .pf files, but it is quite awkward, and we need to fix this kind of thing. ksl

218  if (stype == 0)

220  else if (stype == 2)
221  *spectype = SPECTYPE_UNIFORM; // uniform
222  else if (stype == 3)
223  *spectype = SPECTYPE_POW; // power law
224  else if (stype == 4)
225  *spectype = SPECTYPE_CL_TAB; // broken power law
226  else if (stype == 5)
227  *spectype = SPECTYPE_BREM; // bremstrahlung
228  else
229  {

231  { // Continuing an old model
232  strcpy (model_list, geo.model_list[get_spectype_count]);

239  get_models (model_list, 2, spectype);
