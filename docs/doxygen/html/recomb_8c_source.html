100 fb_topbase_partial (freq)
101  double freq;
102 {
103  int nion;
104  double partial;
105  double x;
106  double gn, gion;
107  double fthresh;
108 
109  fthresh = fb_xtop->freq[0];
110  if (freq < fthresh)
111  return (0.0); // No recombination at frequencies lower than the threshold freq occur
112 
113  nion = fb_xtop->nion;
114 
115  /* JM -- below lines to address bug #195 */
116  gn = 1;
117  if (ion[nion].phot_info > 0) // it's a topbase record
118  gn = config[fb_xtop->nlev].g;
119  else if (ion[nion].phot_info == 0) // it's a VFKY record, so shouldn't really use levels
120  gn = ion[nion].g;
121  else
122  {
123  Error
124  ("fb_topbase_partial: Did not understand cross-section type %i for ion %i. Setting multiplicity to zero!\n",
125  ion[nion].phot_info, nion);
126  gn = 0.0;
127  }
128 
129  gion = ion[nion + 1].g; // Want the g factor of the next ion up
130  x = sigma_phot (fb_xtop, freq);
131  // Now calculate emission using Ferland's expression
132 
133  partial =
134  FBEMISS * gn / (2. * gion) * pow (freq * freq / fbt,
135  1.5) * exp (H_OVER_K * (fthresh -
136  freq) / fbt) * x;
137 
138  // 0=emissivity, 1=heat loss from electrons, 2=photons emissivity
139 
140  if (fbfr == FB_REDUCED)
141  partial *= (freq - fthresh) / freq;
142  else if (fbfr == FB_RATE)
143  partial /= (H * freq);
