514 int
515 select_continuum_scattering_process (kap_cont, kap_es, kap_ff, xplasma)
516  double kap_cont, kap_es, kap_ff;
517  PlasmaPtr xplasma;
518 {
519  int nres;
520  double threshold;
521  double run_tot;
522  int ncont;
523 
524  threshold = random_number (0.0, 1.0) * (kap_cont);
525 
526 
527  /* First check for electron scattering. */
528  if (kap_es > threshold)
529  { /* electron scattering event occurred (SS) */
530  nres = -1; // flag electron scatterin (SS) */
531  }
532  /* Now chech for ff. */
533  else if ((kap_es + kap_ff) > threshold)
534  {
535  nres = -2;
536  }
537  /* Now check for bf. */
538  else
539  {
540 /* use a running sum to find which photoionisation process it was */
541 /*
542 If a non-macro-atom run is being done this part should never be reached.
543 Just do a check that all is well - this can be removed eventually (SS)
544 */
545  if (geo.rt_mode == RT_MODE_2LEVEL)
546  {
547  Error
548  ("calculate_ds: Not using macro atoms but trying to excite one? Aboort.\n");
549  exit (0);
550  }
551  run_tot = kap_es + kap_ff;
552  ncont = 0;
553  while (run_tot < threshold)
554  {
555  run_tot += kap_bf[ncont];
556  ncont++;
557  }
558  /* When it gets here know that excitation is in photoionisation labelled by ncont */
559  nres = NLINES + 1 + xplasma->kbf_use[ncont - 1]; //modified SS Nov 04
560  }
561  return (nres);
562 }
