439 ds_to_cone (cc, p)
440  ConePtr cc;
441  struct photon *p;
442 {
443  double dz, dzdr2;
444  double a, b, c, root[2];
445  double s_to_zero; /* The path length to the xy plane */
446  int i;
447  struct photon pp;
448 
449  /* First of all let's work only in the "northern" hemisphere */
450  stuff_phot (p, &pp);
451 
452  if (pp.x[2] < 0.0)
453  { /*move the photon to the northen hemisphere */
454  pp.x[2] = -pp.x[2];
455  pp.lmn[2] = -pp.lmn[2];
456  }
457 
458  /* Set up and solve the quadratic equation that gives the cone intercept */
459 
460  dzdr2 = cc->dzdr * cc->dzdr;
461  dz = pp.x[2] - cc->z;
462 
463  a = dzdr2 * (pp.lmn[0] * pp.lmn[0] + pp.lmn[1] * pp.lmn[1]) - (pp.lmn[2] * pp.lmn[2]);
464  b = 2. * (dzdr2 * (pp.lmn[0] * pp.x[0] + pp.lmn[1] * pp.x[1]) - pp.lmn[2] * dz);
465  c = dzdr2 * (pp.x[0] * pp.x[0] + pp.x[1] * pp.x[1]) - dz * dz;
466 
467  i = quadratic (a, b, c, root); /* root[i] is the smallest positive root unless i is
468  negative in which case either both roots were negative or both roots were imaginary */
469 
470  /* Calculate the positive path length to the xy plane. If either the
471  photon is travelling in the xy plane or if the intercept is in the negative
472  direction set the path length to infinity */
473 
474  if (pp.x[2] * pp.lmn[2] >= 0)
475  s_to_zero = VERY_BIG;
476  else
477  s_to_zero = (-pp.x[2] / pp.lmn[2]);
478 
479 
480  if (i >= 0 && root[i] < s_to_zero)
481  return (root[i]); /*Because that implies
482  the ray hit the cone before hitting the xy plane */
483  return (s_to_zero);

688 int
689 quadratic (a, b, c, r)
690  double a, b, c, r[];
691 {
692  double q, z;
693 
694  if (a == 0.0)
695  { /* Then it's not really a quadratic but we can solve it
696  anyway */
697  if (b == 0.0)
698  {
699  r[0] = r[1] = -99.;
700  return (-1); /* The roots are extremely imaginary, since both a a b were 0 */
701  }
702 
703  r[0] = r[1] = (-c / b); // Then it was a linear equation. Setting both roots to the same thing could be a problem ksl
704  if (r[0] < 0.0)
705  return (-2); /* Generally speaking we are not interested in
706  negative distances */
707  else
708  return (0);
709  }
710 
711  if ((q = b * b - 4. * a * c) < 0.0)
712  {
713  r[0] = r[1] = -99.;
714  return (-1); /* both roots are imaginary */
715  }
716  q = sqrt (q);
717  z = 0.5 / a;
718 
719  r[0] = (-b - q) * z;
720  r[1] = (-b + q) * z;
721 
722 
723  if (r[0] > 0.0 && (r[0] < r[1] || r[1] <= 0.0))
724  return (0); /* r[0] is smallest positive root */
725  if (r[1] > 0.0 && (r[1] < r[0] || r[0] <= 0.0))
726  return (1); /* r[1] is smallest positive root */
727  return (-2); /* both roots are negative */
728 
729  /* x1 should be the smallest positive root for most applications */
730 }
